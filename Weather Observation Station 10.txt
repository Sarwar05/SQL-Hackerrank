SELECT DISTINCT CITY
FROM STATION
WHERE RIGHT(CITY,1) NOT IN ('A','E','I','O','U');


SELECT DISTINCT CITY
FROM STATION
WHERE (
    CITY NOT LIKE '%A'
    AND 
    CITY NOT LIKE '%E'
    AND 
    CITY NOT LIKE '%I'
    AND 
    CITY NOT LIKE '%O'
    AND 
    CITY NOT LIKE '%U'
)




SELECT DISTINCT CITY
FROM STATION
WHERE NOT(
    CITY  LIKE '%A'
    OR 
    CITY  LIKE '%E'
    OR 
    CITY  LIKE '%I'
    OR 
    CITY  LIKE '%O'
    OR 
    CITY  LIKE '%U'
)

FROM STATION
WHERE SUBSTRING(CITY,LENGTH(CITY),1) NOT IN('A','E','I','O','U')
