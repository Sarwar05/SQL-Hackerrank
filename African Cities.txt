SELECT CITY.NAME
FROM COUNTRY
INNER JOIN CITY 
ON  CITY.CountryCode = COUNTRY.Code
WHERE COUNTRY.CONTINENT = "AFRICA"





SELECT CITY.NAME
FROM COUNTRY,CITY 
WHERE CITY.CountryCode = COUNTRY.Code AND COUNTRY.CONTINENT = "AFRICA"






SELECT NAME
FROM CITY 
WHERE CountryCode IN(
    SELECT Code
    FROM COUNTRY
    WHERE CONTINENT = "AFRICA"
)
