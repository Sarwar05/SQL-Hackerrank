SELECT MAX(SALARY*MONTHS), COUNT(NAME)
FROM EMPLOYEE
WHERE SALARY*MONTHS IN(
    SELECT MAX(SALARY*MONTHS)
    FROM EMPLOYEE
)







SELECT SALARY*MONTHS,COUNT(SALARY*MONTHS)
FROM EMPLOYEE
GROUP BY SALARY*MONTHS
ORDER BY SALARY*MONTHS DESC
LIMIT 1






SELECT (SALARY*MONTHS) AS TOTAL,COUNT(*)
FROM EMPLOYEE
GROUP BY TOTAL
ORDER BY TOTAL DESC
LIMIT 1
