SELECT SUM(CITY.POPULATION)
FROM CITY
INNER JOIN COUNTRY
ON CITY.CountryCode = COUNTRY.Code
WHERE COUNTRY.CONTINENT = 'Asia'










SELECT SUM(CITY.POPULATION)
FROM CITY
INNER JOIN COUNTRY
ON CountryCode = Code
WHERE CONTINENT = 'Asia'










SELECT SUM(CITY.POPULATION)
FROM CITY,COUNTRY
WHERE CONTINENT = 'Asia' AND CODE = COUNTRYCODE









SELECT SUM(POPULATION)
FROM CITY
WHERE COUNTRYCODE IN (
    SELECT CODE
    FROM COUNTRY
    WHERE CONTINENT = 'ASIA'
)
