SELECT DISTINCT CITY
FROM STATION
WHERE (
    CITY LIKE 'A%'
    OR
    CITY LIKE 'E%'
    OR
    CITY LIKE 'I%'
    OR
    CITY LIKE 'O%'
    OR
    CITY LIKE 'U%'
);








SELECT DISTINCT CITY
FROM STATION
WHERE LEFT(CITY,1) IN ('A','E','I','O','U');








SELECT DISTINCT CITY
FROM STATION
WHERE CITY REGEXP '^[AEIOU]';








SELECT DISTINCT CITY
FROM STATION
WHERE SUBSTR(CITY,1,1) IN ('A','E','I','O','U');
